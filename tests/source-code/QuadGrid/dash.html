// define a new iio app
TestDash = function( app, settings ){

  // add a new grid to the app
  app.add(new iio.QuadGrid({
    // set position to app center
    pos: app.center,
    // set width and height to 80px
    width: 80,
    // set line width to 5px
    lineWidth: 5,
    // set number of rows
    R: 3,
    // set number of columns
    C: 3,
    // set outline color to green
    outline: 'green',
    // set line width to 10px
    lineWidth: 10,
    // set dash with 2px length and 5px spacing
    dash: [ 2,5 ]
  }));
}

// start the app on an existing canvas
iio.start( TestDash, 'canvasId' );
