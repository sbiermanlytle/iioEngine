// define a new iio app
TestRadialGradient = function( app, settings ){

	// add a new grid to the app
	app.add(new iio.QuadGrid({
		// set position to app center
		pos: app.center,
		// set width and height to 80px
		width: 80,
		// set line width to 5px
		lineWidth: 5,
		// set number of rows
		R: 3,
		// set number of columns
		C: 3,
		// set color to a new radial gradient
		color: new iio.Gradient({
			// set gradient start position
			start: [ 0, 0 ],
			// set gradient start radius
			startRadius: 1,
			// set gradient end position
			end: [ 0, 0 ],
			// set gradient end radius
			endRadius: 40,
			// define color stops from [0,1]
			stops: [
				// at 0% length, show blue
				[ 0, 'blue' ],
				// at 40% length, show blue
				[ 0.7, 'blue' ],
				// at 100% length, show black
				[ 1, 'black']
			]
		})
	}));
}

// start the app on an existing canvas
iio.start( TestRadialGradient, 'canvasId' );
