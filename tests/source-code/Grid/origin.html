// define a new iio app
TestOrigin = function( app, settings ){

	// add a new grid to the app
	app.add(new iio.Grid({
		// set position relative to app center
		pos: app.center.clone().sub(10,10),
		// set origin of rotation relative to pos
		origin: [10,10],
		// set color to green
		color: 'green',
		// set rotational velocity
		// to .02 radians / update
		rVel: .02,
		// set width and height to 80px
		width: 80,
		// set line width to 5px
		lineWidth: 5,
		// set number of rows
		R: 3,
		// set number of columns
		C: 3
	}));
}

// start the app on an existing canvas
iio.start( TestOrigin, 'canvasId' );
