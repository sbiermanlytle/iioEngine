// define a new iio app
TestOrigin = function( app, settings ){

	// add a new rectangle to the app
	app.add(new iio.Rectangle({
		// set position to app center
		pos: app.center.clone(),
		// set origin of rotation
		origin: [ 8, 8 ],
		// set color to green
		color: 'green',
		// set dimensions to 30x60px
		width: 20,
		height: 30,
		// set rotational velocity
		// to .02 radians / update
		rVel: .02
	}));
}

// start the app on an existing canvas
iio.start( TestOrigin, 'canvasId' );
