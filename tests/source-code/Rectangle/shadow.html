// define a new iio app
TestShadow = function( app, settings ){

	// set app background color to white
	app.set({ color:'white' })

	// add a new rectangle to the app
	app.add(new iio.Rectangle({
		// set position to app center
		pos: app.center.clone(),
		// set color to green
		color: 'green',
		// set dimensions to 30x60px
		width: 30,
		height: 60,
		// set dash with 20px width and spacing
		dash: 20,
		// set line width to 5px
		lineWidth: 5,
		// set shadow color 
		// to 50% transparent black
		shadow: new iio.Color( 0,0,0,.5 ),
		// set shadow blur
		shadowBlur: 5,
		// set shadow offset
		shadowOffset: [ 4,4 ]
	}));
}

// start the app on an existing canvas
iio.start( TestShadow, 'canvasId' );
