// define a new iio app
TestrVelBounds = function( app, settings ){

	// reverse rotational velocity
	function reverse(o){ o.rVel *= -1 }

	// add a new rectangle to the app
	app.add(new iio.Rectangle({
		// set position to app center
		pos: app.center.clone(),
		// set color to green
		color: 'green',
		// set dimensions to 30x60px
		width: 30,
		height: 60,
		// set rotational velocity
		// to .02 radians / update
		rVel: .02,
		// set bounds for movement
		bounds: {
			// define rotation upper limit and callback
			rightRotation: {
				bound: Math.PI/2,
				callback: reverse
			},
			// define rotation lower limit and callback
			leftRotation: {
				bound: 0,
				callback: reverse
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestrVelBounds, 'canvasId' );
