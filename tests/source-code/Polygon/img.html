// define a new iio app
TestImg = function( app, settings ){

	// add a new circle to the app
	app.add(new iio.Circle({
		// set position to app center
		pos: app.center.clone(),
		// set radius
		radius: app.width/2.5,
		// clip the image to a circle shape
		clip: true,
		// attach an image from a URL
		img: 'http://iioengine.com/img/staryNight.jpg'
	}));
}

// start the app on an existing canvas
iio.start( TestImg, 'canvasId' );
