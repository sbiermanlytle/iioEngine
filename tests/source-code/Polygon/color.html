// define a new iio app
TestColor = function( app, settings ){

	// add a new triangle to the app
	app.add(new iio.Polygon({
		// set position to app center
		pos: app.center.clone(),
		// set color to blue
		color: 'blue',
		// define new cycle property
		cycle: 0,
		// set vertices relative to pos
		vs: [
			[   0,-20 ],
			[  20, 20 ],
			[ -20, 20 ]
		],
		// update the color or cycle on each update
		onUpdate: function(){
			switch( this.cycle ){
				case 1: 
					// subtract green if over limit
					if( this.color.g > 100 )
						this.color.g--;
					// subtract red if over limit
					else if( this.color.r > 100 )
						this.color.r--;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
					break;
				case 2: 
					// add blue if under limit
					if( this.color.b < 200 )
						this.color.b++;
					// add red if under limit
					else if( this.color.r < 200 )
						this.color.r++;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
					break;
				case 3: 
					// subtract green if over limit
					if( this.color.g > 0 )
						this.color.g--;
					// subtract red if over limit
					else if( this.color.r > 0 )
						this.color.r--;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
					break;
				default: 
					// add red if under limit
					if( this.color.r < 255 )
						this.color.r++;
					// add blue if under limit
					else if( this.color.b < 255 )
						this.color.b++;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestColor, 'canvasId' );
