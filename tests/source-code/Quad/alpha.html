// define a new iio app
TestAlpha = function( app, settings ){

	// add a new quad to the app
	app.add(new iio.Quad({
		// set position to app center
		pos: app.center.clone(),
		// set color to green
		color: 'green',
		// set dimensions to 30x60px
		width: 30,
		height: 60,
		// define fade animation
		fade: {
			// set fade speed
			// to .03 alpha / update
			speed: .03,
			// add lower bound to prevent
			// full transparency
			lowerBound: .2,
			// callback when alpha = 0 or 1
			callback: function(o){
				// reverse fade direction
				o.fade.speed *= -1;
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestAlpha, 'canvasId' );
