// define a new iio app
TestDash = function( app, settings ){

	// add a new text to the app
	app.add(new iio.Text({
		// set position to app center
		pos: app.center,
		// set text to iio
		text: 'iio',
		// set outline color to green
		outline: 'green',
		// set line width to 1px
		lineWidth: 1,
		// set dash with 2px length and spacing
		dash: 2
	}));
}

// start the app on an existing canvas
iio.start( TestDash, 'canvasId' );
