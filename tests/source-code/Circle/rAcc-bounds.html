// define a new iio app
TestrAccBounds = function( app, settings ){

	// add a new circle to the app
	app.add(new iio.Circle({
		// set position to app center
		pos: app.center.clone(),
		// set rotation origin relative to pos
		origin: [ 12, -12 ],
		// set color to green
		color: 'green',
		// set radius to 25px
		radius: 25,
		// set rotational acceleration
		// to .0015 radians / update
		rAcc: .0015,
		// set bounds for movement
		bounds: {
			// define rotation upper limit and callback
			rightRotation: {
				bound: Math.PI/2,
				callback: function(o){
					// reverse rotational acceleration
					o.rAcc *= -1; 
					// set rotational velocity
					o.rVel = -.01;
				}
			},
			// define rotation lower limit and callback
			leftRotation: {
				bound: -Math.PI/2, 
				callback: function(o){
					// reverse rotational acceleration
					o.rAcc *= -1; 
					// set rotational velocity
					o.rVel = .01;
				}
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestrAccBounds, 'canvasId' );
