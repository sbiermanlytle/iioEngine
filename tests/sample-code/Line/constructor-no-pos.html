// define a new iio app
TestConstructorNoPos = function( app, settings ){

	// add a new line to the app
	app.add(new iio.Line({
		// set line color to blue
		color: 'blue',
		// set line width to 10px
		width: 10,
		// set line vertices relative to app 0,0
		vs:[
			// start vertex
			[ 20, 20 ],
			// end vertex
			[ app.width-20, app.height-20 ]
		]
	}));
}

// start the app on an existing canvas
iio.start( TestConstructorNoPos, 'canvasId' );
