TestBezierAccs = function( app, settings ){

	var speed = 1;

	var checkBound = function( o, v ){
		if(o.bezier[v].x > app.width 
		 || o.bezier[v].x < -app.width)
			o.bezierVels[v].x *= -1;
	}

	app.add(new iio.Line({
		pos: app.center,
		color: 'green',
		width: 10,
		vs:[
			[ 0,-30 ],
			[ 0, 30 ]
		],
		bezier: [
			[  app.width, 0 ],
			[ -app.width, 0 ]
		],
		bezierVels: [
			[ -speed, 0 ],
			[  speed, 0 ]
		],
		onUpdate: function(){
			checkBound( this, 0 );
			checkBound( this, 1 );
		}
	}));
}

iio.start( TestBezierAccs, 'canvasId' );
