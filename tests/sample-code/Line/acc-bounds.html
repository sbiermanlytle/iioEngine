// define a new iio app
TestAccBounds = function( app, settings ){

	// define a speed variable
	var speed = 0.4;

	// add a new line to the app
	app.add(new iio.Line({
		// set position to app center
		pos: app.center.clone(),
		// set color to green
		color: 'green',
		// set width to 10px
		width: 10,
		// set x velocity
		vel: [ speed, 0 ],
		// set x acceleration
		acc: [ .01, 0 ],
		// set vertices relative to pos
		vs:[
			// start vertex position
			[ 0,-30 ],
			// end vertex position
			[ 0, 30 ]
		],
		// set bounds for movement
		bounds: {
			// define pos.x upper limit and callback
			right: {
				bound: app.center.x+15, 
				callback: function(o){
					// set the x velocity
					o.vel.x = -.5
				} 
			},
			// define pos.x lower limit and callback
			left: {
				bound: app.center.x-15, 
				callback: function(o){
					// set the x velocity
					o.vel.x = .5
				} 
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestAccBounds, 'canvasId' );
