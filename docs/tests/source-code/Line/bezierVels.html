// define a new iio app
TestBezierVels = function( app, settings ){

  // define a speed variable
  var speed = 1;

  // reverse the velocity of the bezier handle
  // if its position is outside the app screen
  // bezier handle positions are relative to pos
  var checkBound = function( o,v ){
    if(o.bezier[v].x > app.width 
     || o.bezier[v].x < -app.width)
       // reverse x velocity
      o.bezierVels[v].x *= -1;
  }

  // add a new line to the app
  app.add(new iio.Line({
    // set position to app center
    pos: app.center,
    // set color to blue
    color: 'blue',
    // set width to 10px
    width: 10,
    // set vertices relative to pos
    vs:[
      // start vertex position
      [ 0,-30 ],
      // end vertex position
      [ 0, 30 ]
    ],
    // set bezier handle positions
    bezier: [
      // first bezier handle position
      [  app.width, 0 ],
      // second bezier handle position
      [ -app.width, 0 ]
    ],
    // set bezier handle velocities
    bezierVels: [
      // first bezier handle velocity
      [ -speed, 0 ],
      // second bezier handle velocity
      [  speed, 0 ]
    ],
    // check the position of each bezier handle
    // on every update
    onUpdate: function(){
      // check first bezier handle position
      checkBound( this, 0 );
      // second bezier handle position
      checkBound( this, 1 );
    }
  }));
}

// start the app on an existing canvas
iio.start( TestBezierVels, 'canvasId' );
