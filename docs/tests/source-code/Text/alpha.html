// define a new iio app
TestAlpha = function( app, settings ){

  // add a new text to the app
  app.add(new iio.Text({
    // set position to app center
    pos: app.center,
    // set color to green
    color: 'green',
    // set text to iio
    text: 'iio',
    // define fade animation
    fade: {
      // set fade speed
      // to .03 alpha / update
      speed: .03,
      // set lower bound for alpha
      lowerBound: .2,
      // callback when alpha = 0 or 1
      callback: function(o){
        // reverse fade direction
        o.fade.speed *= -1;
      }
    }
  }));
}

// start the app on an existing canvas
iio.start( TestAlpha, 'canvasId' );
