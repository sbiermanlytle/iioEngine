// define a new iio app
TestFlip = function( app, settings ){
  
  // set the app update rate to 1 fps
  app.loop(1);

  // add a new text to the app
  app.add(new iio.Text({
    // set position to app center
    pos: app.center,
    // set text to iio
    text: 'iio',
    // flip the object horizontally
    flip: 'x',
    // flip the image on each update
    onUpdate: function(){
      if(this.flip == 'x')
        // no flip
        this.flip = false;
        // horizontal flip
      else this.flip = 'x';
    }
  }));
}

// start the app on an existing canvas
iio.start( TestFlip, 'canvasId' );
