// define a new iio app
TestVelBounds = function( app, settings ){
  
  // define a speed variable
  var speed = 1;

  // reverse the given objects x velocity
  function reverse(o){ o.vel.x *= -1 }

  // add a new text to the app
  app.add(new iio.Text({
    // set position to app center
    pos: app.center.clone(),
    // set color to blue
    color: 'blue',
    // set text to iio
    text: 'iio',
    // set x velocity
    vel: [ speed, 0 ],
    // set bounds for movement
    bounds: {
      // define pos.x upper limit and callback
      right: {
        bound: app.width,
        callback: reverse
      },
      // define pos.x lower limit and callback
      left: {
        bound: 0, 
        callback: reverse
      }
    }
  }));
}

// start the app on an existing canvas
iio.start( TestVelBounds, 'canvasId' );
