// define a new iio app
TestOrigin = function( app, settings ){

  // add a new triangle to the app
  app.add(new iio.Polygon({
    // set origin of rotation
    origin: app.center,
    // set color to green
    color: 'green',
    // set rotational velocity
    // to .02 radians / update
    rVel: .02,
    // set vertices relative to app 0,0
    vs:[
      [ 30, 15 ],
      [ 15, 45 ],
      [ 45, 45 ]
    ]
  }));
}

// start the app on an existing canvas
iio.start( TestOrigin, 'canvasId' );
